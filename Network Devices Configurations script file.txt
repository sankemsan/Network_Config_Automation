def load_configs(file_path):
    configs = {}
    current_device = None
    with open(file_path, "r") as f:
        for line in f:
            line = line.strip()
            if line.startswith("hostname"):
                current_device = line.split()[1]
                configs[current_device] = []
            elif current_device:
                configs[current_device].append(line)
    return configs
